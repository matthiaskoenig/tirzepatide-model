from typing import Dict

from sbmlsim.data import DataSet, load_pkdb_dataframe
from sbmlsim.fit import FitMapping, FitData
from sbmlutils.console import console

from pkdb_models.models.tirzepatide.experiments.base_experiment import (
    TirzepatideSimulationExperiment,
)
from pkdb_models.models.tirzepatide.experiments.metadata import (
    Tissue,
    Route,
    Dosing,
    ApplicationForm,
    Health,
    Fasting,
    TirzepatideMappingMetaData,
    Coadministration,
)

from sbmlsim.plot import Axis, Figure
from sbmlsim.simulation import Timecourse, TimecourseSim

from pkdb_models.models.tirzepatide.helpers import run_experiments


class Coskun2018(TirzepatideSimulationExperiment):
    """Simulation experiment of Coskun2018."""

    bodyweights = {
        "SAD": 71.9,  # kg
        "MAD": 71.1,  # kg
        "POC": 86.0,  # kg
    }

    # Dosing schemes for SAD, MAD, and POC cohorts
    doses_sad = [0.25, 0.5, 1, 2.5, 5, 8]  # mg
    doses_mad = [0.5, 1.5, 4.5, 10]  # mg
    doses_poc = [0.5, 5, 10, 15]  # mg

    colors = {
        0.25: "crimson",
        0.5: "goldenrod",
        1: "limegreen",
        2.5: "steelblue",
        5: "indigo",
        8: "saddlebrown",
        1.5: "darkorange",
        4.5: "purple",
        10: "deepskyblue",
        15: "darkred"
    }

    def datasets(self) -> Dict[str, DataSet]:
        dsets = {}

        for fig_id in ["Fig1_EMA", "Fig1_FDA", "Fig2_FDA"]:
            df = load_pkdb_dataframe(f"{self.sid}_{fig_id}", data_path=self.data_path)
            for label, df_label in df.groupby("label"):
                if label.startswith("tirzepatide_"):
                    dset = DataSet.from_df(df_label, self.ureg)
                    dset.unit_conversion("mean", 1 / self.Mr.tir)
                    dsets[label] = dset
        return dsets

    def simulations(self) -> Dict[str, TimecourseSim]:
        Q_ = self.Q_
        tcsims = {}

        for cohort, doses in zip(["SAD", "MAD", "POC"], [self.doses_sad, self.doses_mad, self.doses_poc]):
            for dose in doses:
                tcsims[f"po_tir_{cohort.lower()}_{dose}"] = TimecourseSim(
                    [Timecourse(
                        start=0,
                        end=14 * 7 * 24 * 60,
                        steps=2000,
                        changes={
                            **self.default_changes(),
                            "BW": Q_(self.bodyweights[cohort], "kg"),
                            "SCDOSE_tir": Q_(dose, "mg"),
                        },
                    )]
                )

        return tcsims

    def fit_mappings(self) -> Dict[str, FitMapping]:
        mappings = {}
        for cohort, doses in zip(["SAD", "MAD", "POC"], [self.doses_sad, self.doses_mad, self.doses_poc]):
            for dose in doses:
                dose_str = str(dose).replace(".", "_")
                mappings[f"fm_po_tir_{cohort.lower()}_{dose}"] = FitMapping(
                    self,
                    reference=FitData(
                        self,
                        dataset=f"tirzepatide_{cohort.lower()}{dose_str}",
                        xid="time",
                        yid="mean",
                        yid_sd="mean_sd",
                        count="count",
                    ),
                    observable=FitData(
                        self, task=f"task_po_tir_{cohort.lower()}_{dose}", xid="time", yid=f"[Cve_tir]",
                    ),
                    metadata=TirzepatideMappingMetaData(
                        tissue=Tissue.PLASMA,
                        route=Route.PO,
                        application_form=ApplicationForm.SUBCUTANEOUS,
                        dosing=Dosing.MULTIPLE if cohort in ["MAD", "POC"] else Dosing.SINGLE,
                        health=Health.HEALTHY if cohort in ["SAD", "MAD"] else Health.T2DM,
                        fasting=Fasting.FASTED,
                        coadministration=Coadministration.NONE,
                    ),
                )

        return mappings

    def figures(self) -> Dict[str, Figure]:
        figure_ids = {"SAD": "Fig1_EMA", "MAD": "Fig1_FDA", "POC": "Fig2_FDA"}
        figures = {}

        for cohort, doses in zip(["SAD", "MAD", "POC"], [self.doses_sad, self.doses_mad, self.doses_poc]):
            fig_id = figure_ids[cohort]
            fig = Figure(
                experiment=self,
                sid=fig_id,
                num_rows=1,
                name=f"{self.__class__.__name__} ({cohort})",
            )
            plots = fig.create_plots(
                xaxis=Axis(self.label_time, unit="week", min=-0.5, max=6), legend=True
            )
            plots[0].set_yaxis(self.label_tir, unit=self.unit_tir)

            for dose in doses:
                dose_str = str(dose).replace(".", "_")
                plots[0].add_data(
                    task=f"task_po_tir_{cohort.lower()}_{dose}",
                    xid="time",
                    yid="[Cve_tir]",
                    label=f"Sim {dose} mg tir ({cohort})",
                    color=self.colors.get(dose, "black"),
                )
                plots[0].add_data(
                    dataset=f"tirzepatide_{cohort.lower()}{dose_str}",
                    xid="time",
                    yid="mean",
                    yid_sd="mean_sd",
                    count="count",
                    label=f"Data {dose} mg ({cohort})",
                    color=self.colors.get(dose, "black"),
                )

            figures[fig.sid] = fig

        return figures


if __name__ == "__main__":
    run_experiments(Coskun2018, output_dir=Coskun2018.__name__)
