from typing import Dict

from sbmlsim.data import DataSet, load_pkdb_dataframe
from sbmlsim.fit import FitMapping, FitData
from sbmlutils.console import console

from pkdb_models.models.tirzepatide.experiments.base_experiment import (
    TirzepatideSimulationExperiment,
)
from pkdb_models.models.tirzepatide.experiments.metadata import Tissue, Route, Dosing, ApplicationForm, Health, \
    Fasting, TirzepatideMappingMetaData, Coadministration

from sbmlsim.plot import Axis, Figure
from sbmlsim.simulation import Timecourse, TimecourseSim

from pkdb_models.models.tirzepatide.helpers import run_experiments

class Urva2022(TirzepatideSimulationExperiment):
    """Simulation experiment of Urva2022."""

    bodyweight = {
        "normal"    : 96.0,
        "mild"      : 82.75,
        "moderate"  : 95.12,
        "severe"    : 88.56,
    }

    groups = list(bodyweight.keys())

    kidney_function = {
        "normal": TirzepatideSimulationExperiment.cirrhosis_map["Control"],
        "mild": TirzepatideSimulationExperiment.cirrhosis_map["Mild cirrhosis"],
        "moderate": TirzepatideSimulationExperiment.cirrhosis_map["Moderate cirrhosis"],
        "severe": TirzepatideSimulationExperiment.cirrhosis_map["Severe cirrhosis"]
    }
    individual_keys = [
        'tirzepatide_M1', 'tirzepatide_M2', 'tirzepatide_M3', 'tirzepatide_M4', 'tirzepatide_M5', 'tirzepatide_M6',
        'tirzepatide_MO1', 'tirzepatide_MO2', 'tirzepatide_MO3', 'tirzepatide_MO4', 'tirzepatide_MO5', 'tirzepatide_MO6',
        'tirzepatide_N1', 'tirzepatide_N10', 'tirzepatide_N11', 'tirzepatide_N12', 'tirzepatide_N13', 'tirzepatide_N2', 'tirzepatide_N3',
        'tirzepatide_N4', 'tirzepatide_N5', 'tirzepatide_N6', 'tirzepatide_N7', 'tirzepatide_N8', 'tirzepatide_N9',
        'tirzepatide_S1', 'tirzepatide_S2', 'tirzepatide_S3', 'tirzepatide_S4', 'tirzepatide_S5', 'tirzepatide_S6', 'tirzepatide_S7'
    ]

    def group_individual_keys(self) -> Dict[str, list]:
        """Group and sort individual keys by their category based on prefix."""
        groups = {"normal": [], "mild": [], "moderate": [], "severe": []}
        for key in self.individual_keys:
            if key.startswith("tirzepatide_MO"):
                groups["moderate"].append(key)
            elif key.startswith("tirzepatide_M"):
                groups["mild"].append(key)
            elif key.startswith("tirzepatide_N"):
                groups["normal"].append(key)
            elif key.startswith("tirzepatide_S"):
                groups["severe"].append(key)
        for grp, keys in groups.items():
            groups[grp] = sorted(
                keys, key=lambda k: int(''.join(filter(str.isdigit, k))) if ''.join(filter(str.isdigit, k)) else 0
            )
        return groups

    def datasets(self) -> Dict[str, DataSet]:
        dsets = {}
        for fig_id in ["Fig1A", "Fig1"]:
            df = load_pkdb_dataframe(f"{self.sid}_{fig_id}", data_path=self.data_path)
            for label, df_label in df.groupby("label"):
                if label.startswith("tirzepatide_"):
                    dset = DataSet.from_df(df_label, self.ureg)
                    if fig_id == "Fig1A":
                        dset.unit_conversion("mean", 1 / self.Mr.tir)
                    else:
                        dset.unit_conversion("value", 1 / self.Mr.tir)
                    dsets[f"{label}"] = dset
        return dsets

    def simulations(self) -> Dict[str, TimecourseSim]:
        Q_ = self.Q_
        tcsims = {}

        for group, value in self.kidney_function.items():
            tcsims[f"tir_hepatic_{group}"] = TimecourseSim(
                Timecourse(
                    start=0,
                    end=5 * 7 * 24 * 60,  # 4 weeks
                    steps=2000,
                    changes={
                        **self.default_changes(),
                        "BW": Q_(self.bodyweight[group], "kg"),
                        "SCDOSE_tir": Q_(5, "mg"),
                        "f_cirrhosis": Q_(value, "dimensionless")
                    },
                )
            )
        grouped_keys = self.group_individual_keys()
        for group, keys in grouped_keys.items():
            for key in keys:
                tcsims[f"tir_hepatic_{key}"] = TimecourseSim(
                    Timecourse(
                        start=0,
                        end=4 * 7 * 24 * 60,  # 4 weeks
                        steps=2000,
                        changes={
                            **self.default_changes(),
                            "BW": Q_(self.bodyweight[group], "kg"),
                            "SCDOSE_tir": Q_(5, "mg"),
                            "f_cirrhosis": Q_(self.kidney_function[group], "dimensionless"),
                        },
                    )
                )
        return tcsims

    def fit_mappings(self) -> Dict[str, FitMapping]:
        mappings = {}
        for group in self.groups:
            mappings[f"fm_tir_hepatic_{group}"] = FitMapping(
                self,
                reference=FitData(
                    self,
                    dataset=f"tirzepatide_{group}",
                    xid="time",
                    yid="mean",
                    yid_sd="mean_sd",
                    count="count",
                ),
                observable=FitData(
                    self, task=f"task_tir_hepatic_{group}", xid="time", yid=f"[Cve_tir]",
                ),
                metadata=TirzepatideMappingMetaData(
                    tissue=Tissue.PLASMA,
                    route=Route.PO,
                    application_form=ApplicationForm.SUBCUTANEOUS,
                    dosing=Dosing.SINGLE,
                    health=Health.T2DM if group == "normal" else Health.HEPATIC_IMPAIRMENT,
                    fasting=Fasting.NR,
                    coadministration=Coadministration.NONE,
                ),
            )
        grouped_keys = self.group_individual_keys()
        for group, keys in grouped_keys.items():
            for key in keys:
                mappings[f"fm_tir_hepatic_{key}"] = FitMapping(
                    self,
                    reference=FitData(
                        self,
                        dataset=key,
                        xid="time",
                        yid="value",
                        count="count",
                    ),
                    observable=FitData(
                        self, task=f"task_tir_hepatic_{key}", xid="time", yid="[Cve_tir]",
                    ),
                    metadata=TirzepatideMappingMetaData(
                        tissue=Tissue.PLASMA,
                        route=Route.PO,
                        application_form=ApplicationForm.SUBCUTANEOUS,
                        dosing=Dosing.SINGLE,
                        health=Health.T2DM if group == "normal" else Health.HEPATIC_IMPAIRMENT,
                        fasting=Fasting.NR,
                        coadministration=Coadministration.NONE,
                    ),
                )
        return mappings

    def figures(self) -> Dict[str, Figure]:
        fig = Figure(
            experiment=self,
            sid=f"Fig1A",
            name=f"{self.__class__.__name__} (hepatic impairment)",
        )
        fig2 = Figure(
            experiment=self,
            sid=f"Fig1",
            num_cols=2,
            num_rows=2,
            name=f"{self.__class__.__name__} (hepatic impairment, individuals)",
        )
        plots = fig.create_plots(
            xaxis=Axis(self.label_time, unit="week"),
            yaxis=Axis(self.label_tir, unit=self.unit_tir),
            legend=True)
        plots2 = fig2.create_plots(
            xaxis=Axis(self.label_time, unit="week"),
            yaxis=Axis(self.label_tir, unit=self.unit_tir),
            legend=True)
        # simulations
        for group, color in zip(self.groups, self.cirrhosis_colors.values()):
            plots[0].add_data(
                task=f"task_tir_hepatic_{group}",
                xid="time",
                yid="[Cve_tir]",
                label=group,
                color=color,
            )
        # data
        for group, color in zip(self.groups, self.cirrhosis_colors.values()):
            plots[0].add_data(
                dataset=f"tirzepatide_{group}",
                xid="time",
                yid="mean",
                yid_sd="mean_sd",
                count="count",
                label=f"{group}",
                color=color,
            )
        # Individual
        grouped_keys = self.group_individual_keys()
        for i, ((group, keys), color) in enumerate(zip(grouped_keys.items(),
                                                       self.cirrhosis_colors.values())):
            for key in keys:
                plots2[i].add_data(
                    task=f"task_tir_hepatic_{group}",
                    xid="time",
                    yid="[Cve_tir]",
                    # label=group,
                    color=color,
                )
                plots2[i].add_data(
                    dataset=key,
                    xid="time",
                    yid="value",
                    count="count",
                    # label=key,
                    color=color,
                )
        return {
            fig.sid: fig,
            fig2.sid: fig2
        }

if __name__ == "__main__":
    run_experiments(Urva2022, output_dir=Urva2022.__name__)
